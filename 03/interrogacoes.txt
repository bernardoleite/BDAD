7- Qual a sala mais usada para se realizar exames?
	(Qual a sala que realiza o mais tipos de exame?)
	
SELECT SalaExame FROM SalaTipo
GROUP BY SalaExame
HAVING COUNT(*) = (
                   SELECT MAX(Cnt) 
                   FROM(
                         SELECT COUNT(*) as Cnt
                         FROM SalaTipo
                         GROUP BY SalaExame

                    );
